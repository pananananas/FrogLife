220 private void updatePosition() {
221
222 if (!left && !right && !up && !down) {
223 playerMoving = false;
224 return;
225 }
226 float xSpeed = 0;
227 float ySpeed = 0;
228
229 if (left && !right) {
230 xSpeed = -playerSpeed;
231 flipX = width;
232 flipW = -1;
233 }
234 else if (right && !left) {
235 xSpeed = playerSpeed;
236 flipX = 0;
237 flipW = 1;
238 }
239 if (up && !down)
240 ySpeed = -playerSpeed;
241 else if (down && !up)
242 ySpeed = playerSpeed;
243
244 if (xSpeed != 0 && ySpeed != 0) {
245 xSpeed /= 1.414f;
246 ySpeed /= 1.414f;
247 }
248
249 if (canMoveHere(hitbox.x + xSpeed, hitbox.y, hitbox.width, hitbox.height, lvlData)) {
250 hitbox.x += xSpeed;
251 playerMoving = true;
252 }
253 if (canMoveHere(hitbox.x, hitbox.y + ySpeed, hitbox.width, hitbox.height, lvlData)) {
254 hitbox.y += ySpeed;
255 playerMoving = true;
256 }
